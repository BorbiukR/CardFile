CREATE DATABASE [DBname]		-- створює таблицю
USE [DBname]; 				-- використання таблиці
DROP DATABASE [DBname]			-- видаляє таблицю

CREATE TABLE Customers
(
    Id INT PRIMARY KEY IDENTITY,				-- робить праймері ключем
    Age INT DEFAULT 18 CHECK(Age >0 AND Age < 100), 		-- DEFAULT встановлює значення 18, якщо не було нічого вказано	
								-- CHECK задає параметри перевірки					
    FirstName NVARCHAR(20),
    LastName NVARCHAR(20) NULL,		-- може бути нуль
    Email VARCHAR(30) NOT NULL,		--не може бути нуль
    Phone VARCHAR(20) UNIQUE,   	-- робиться унікальним, два значення не можна встановити
    PRIMARY KEY(Id)			-- ставиться праймері ключем
)

CREATE TABLE Orders				-- звязвки в SQL
(
    Id INT PRIMARY KEY IDENTITY,
    CustomerId INT REFERENCES Customers (Id),
    CreatedAt Date
);
